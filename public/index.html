API Endpoints

"GET" /api/topics

--get all the topics

"GET" /api/topics/:topic_slug/articles

--Return all the articles for a certain topic, e.g: /api/topics/football/articles

"POST" /api/topics/:topic_slug/articles

--Add a new article to a topic. This route requires a JSON body with title and body key value pairs e.g: { "title": "new article", "body": "This is my new article content"}

"GET" /api/articles

--get all the articles

"GET" /api/articles/:article_id

--get an individual article

"GET" /api/articles/:article_id/comments

--get all comments for an individual article

"POST" /api/articles/:article_id/comments

--Add a new comment to an article. This route requires a JSON body with body and created_by key value pairs e.g: {"body": "This is my new comment", "created_by": 'mongo id' for a user}

"PUT" /api/articles/:article_id

--Increment or Decrement the votes of an article by one. This route requires a vote query of 'up' or 'down' e.g: /api/articles/:article_id?vote=up

"PUT" /api/comments/:comment_id

--Increment or Decrement the votes of an comment by one. This route requires a vote query of 'up' or 'down' e.g: /api/articles/:article_id?vote=up

"DELETE" /api/comments/:comment_id

--Deletes a comment

"GET" /api/users

--get all the users

"GET" /api/users/:username

--returns a single users data
